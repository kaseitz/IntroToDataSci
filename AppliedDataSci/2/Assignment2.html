An NOAA dataset has been stored in the file data/C2A2_data/BinnedCsvs_d400/fb441e62df2d58994928907a91895ec62c2c42e6cd075c2700843b89.csv. This is the dataset to use for this assignment. Note: The data for this assignment comes from a subset of The National Centers for Environmental Information (NCEI) Daily Global Historical Climatology Network (GHCN-Daily). The GHCN-Daily is comprised of daily climate records from thousands of land surface stations across the globe.

Each row in the assignment datafile corresponds to a single observation.

The following variables are provided to you:

  • id : station identification code
  • date : date in YYYY-MM-DD format (e.g. 2012-01-24 = January 24, 2012)
  • element : indicator of element type
    • TMAX : Maximum temperature (tenths of degrees C)
    • TMIN : Minimum temperature (tenths of degrees C)
  • value : data value for element (tenths of degrees C)

def read_data():
    filename = 'data/C2A2_data/BinnedCsvs_d400/fb441e62df2d58994928907a91895ec62c2c42e6cd075c2700843b89.csv' 
    df = pd.read_csv(filename).rename(columns = {'Data_Value': "Temp"}) 
    df['Temp'] = df['Temp'].apply(lambda degrees: degrees / 10)
    
    #remove leapdays from the file
    df['Date'] = pd.to_datetime(df['Date'])
    df = df[~((df.Date.dt.month == 2) & (df.Date.dt.day == 29))]
    df['Year'] = pd.DatetimeIndex(df['Date']).year
    df['Month'] = pd.DatetimeIndex(df['Date']).month
    df['Day'] = pd.DatetimeIndex(df['Date']).day
    return  df

def aggregate_data(df):
    #split into a min and max df
    df_max = df[df['Element'] == 'TMAX']
    df_min = df[df['Element'] == 'TMIN']
    #for each day find the record high and low
    df_max = df_max.groupby(['Month', 'Day'])['Temp'].max().rename('max')
    df_min = df_min.groupby(['Month', 'Day'])['Temp'].min().rename('min')
    df_merged = pd.concat([df_max, df_min], axis = 1)
    df_merged = df_merged.reset_index()
    return df_merged

def make_plot():
    df, df_2015 = plotting_data()
    record_high = df_2015[df_2015['max'] > df['max']]
    record_low = df_2015[df_2015['min'] < df['min']]
    
    plt.rcParams["figure.figsize"] = (15,7)
    ax = plt.gca()
    df.plot(kind='line',x=df.index ,y='max',ax=ax, color = "#F784A8")
    df.plot(kind='line',x=df.index ,y='min',ax=ax, color = "#91c6e6")
    #Overlay a scatter of the 2015 data for any points (highs and lows) 
    #for which the ten year record (2005-2014) record high or record low was broken in 2015.
    plt.scatter(record_high.index, record_high['max'], color='red', s=10)
    plt.scatter(record_low.index, record_low['min'], color='blue', s=8)
    #fill between the min and max line
    plt.fill_between(df.index, df['min'], df['max'], color = "#f4f0f0")
    
    # Create legend and title
    # loc=0 provides the best position for the legend
    plt.legend(['Record high (2005-2014)','Record low (2005-2014)','Record breaking high (2015)',
                'Record breaking low (2015)'],loc=0,frameon=False)
    months = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'June', 'July', 'Aug', 'Sep',
             'Oct', 'Nov', 'Dec']
    month_days = [1, 32, 60, 91, 121, 152, 182, 213, 243, 274, 304, 335]
    plt.xticks(month_days, months)
    
    # Set axis names and title:
    plt.xlabel('Month', fontsize=10)
    plt.ylabel(' Celsius', fontsize=10)
    plt.title('Temperature in Ann Arbour, Michigan (2005-2015)', fontsize=12)
    
    plt.show()
    plt.clf()
    plt.cla()
    plt.close()
    return
    
make_plot()
